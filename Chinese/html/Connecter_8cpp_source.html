42const QString CConnecter::Id()
43{
44 QString szId = Protocol() + "_" + GetPlugClient()->Name();
45 if(GetParameter())
46 {
47 if(!GetParameter()->GetName().isEmpty())
48 szId += "_" + GetParameter()->GetName();
49 if(!GetParameter()->m_Net.GetHost().isEmpty())

52 }

153int CConnecter::OpenDialogSettings(QWidget *parent)
154{
155 int nRet = -1;
156 QDialog* p = OnOpenDialogSettings(parent);
157 if(p)
158 {
159 // The dialog is closed when the connect is close.
160 bool check = connect(this, SIGNAL(sigDisconnected()),
161 p, SLOT(reject()));
162 Q_ASSERT(check);
163 p->setWindowIcon(this->Icon());
164 p->setWindowTitle(tr("Set ") + GetPlugClient()->DisplayName());
165#ifdef BUILD_QUIWidget
166 QUIWidget* quiwidget = new QUIWidget();
167 quiwidget->setMainWidget(p);
168 check = connect(p, SIGNAL(accepted()), quiwidget, SLOT(accept()));
169 Q_ASSERT(check);
170 check = connect(p, SIGNAL(rejected()), quiwidget, SLOT(reject()));
171 Q_ASSERT(check);
172 p = quiwidget;
173#endif
174
175 p->setAttribute(Qt::WA_DeleteOnClose);
176 nRet = p->exec();
177 } else {
178 qCritical(log) << "The Protocol [" << Protocol() << "] don't settings dialog";
179 }
180 return nRet;
181}

207int CConnecter::SetParameterClient(CParameterClient* pPara)

229 "If you are sure the parameter of connecter does not need CParameterClient. "
230 "please overload the SetParameterClient in the"
231 << metaObject()->className() << ". don't set it";
232 Q_ASSERT(false);

251int CConnecter::SetParameter(CParameterBase *p)
252{
253 if(m_pParameter)
254 m_pParameter->disconnect(this);
255
256 m_pParameter = p;
257
258 if(GetParameter())
259 {
260 bool check = false;
261 check = connect(GetParameter(), SIGNAL(sigNameChanged()),
262 this, SLOT(slotUpdateName()));
263 Q_ASSERT(check);
264 check = connect(GetParameter(), SIGNAL(sigShowServerNameChanged()),
265 this, SLOT(slotShowServerName()));
266 Q_ASSERT(check);
267 check = connect(GetParameter(), &CParameter::sigChanged,
268 this, [&](){
269 emit this->sigUpdateParameters(this);
270 });
271 Q_ASSERT(check);
272 }
273 return 0;
