72CConnect::OnInitReturnValue CConnectFreeRDP::OnInit()
73{
74 qDebug(log) << "CConnectFreeRdp::OnInit()";
75 int nRet = 0;
76
77 m_writeEvent = CreateEvent(NULL, TRUE, FALSE, NULL);
78 if(!m_writeEvent)
79 qCritical(log) << "CreateEvent failed";
80
81 ZeroMemory(&m_ClientEntryPoints, sizeof(RDP_CLIENT_ENTRY_POINTS));
82 m_ClientEntryPoints.Version = RDP_CLIENT_INTERFACE_VERSION;
83 m_ClientEntryPoints.Size = sizeof(RDP_CLIENT_ENTRY_POINTS);
84 //m_ClientEntryPoints.settings = m_pParameter->m_pSettings;
85 m_ClientEntryPoints.GlobalInit = cbGlobalInit;
86 m_ClientEntryPoints.GlobalUninit = cbGlobalUninit;
87 m_ClientEntryPoints.ClientNew = cbClientNew;
88 m_ClientEntryPoints.ClientFree = cbClientFree;
89 m_ClientEntryPoints.ClientStart = cbClientStart;
90 m_ClientEntryPoints.ClientStop = cbClientStop;
91 m_ClientEntryPoints.ContextSize = sizeof(ClientContext);
92
93 auto pRdpContext = freerdp_client_context_new(&m_ClientEntryPoints);
94 if(pRdpContext)
95 {
96 m_pContext = (ClientContext*)pRdpContext;
97 m_pContext->pThis = this;
98 } else {
99 qCritical(log) << "freerdp_client_context_new fail";
100 return OnInitReturnValue::Fail;
101 }
102
103 rdpSettings* settings = pRdpContext->settings;
104 if(!settings) {
105 qCritical(log) << "settings is null";
106 return OnInitReturnValue::Fail;
107 }
108
109#if FreeRDP_VERSION_MAJOR >= 3
110 if (!stream_dump_register_handlers(pRdpContext,
111 CONNECTION_STATE_MCS_CREATE_REQUEST,
112 FALSE))
113 return OnInitReturnValue::Fail;
114#endif
115
116 auto &user = m_pParameter->m_Net.m_User;
117 if(!user.GetUser().isEmpty())
118 freerdp_settings_set_string(
119 settings, FreeRDP_Username,
120 user.GetUser().toStdString().c_str());
121 if(!user.GetPassword().isEmpty())
122 freerdp_settings_set_string(
123 settings, FreeRDP_Password,
124 user.GetPassword().toStdString().c_str());
125
126 freerdp_settings_set_bool(
127 settings, FreeRDP_RedirectClipboard, m_pParameter->GetClipboard());
128
129#if FreeRDP_VERSION_MAJOR >= 3
130 bool bOnlyView = m_pParameter->GetOnlyView();
131 freerdp_settings_set_bool(
132 settings, FreeRDP_SuspendInput, bOnlyView);
133#endif
134
135 freerdp_settings_set_uint32(settings, FreeRDP_DesktopWidth,
136 m_pParameter->GetDesktopWidth());
137 freerdp_settings_set_uint32(settings, FreeRDP_DesktopHeight,
138 m_pParameter->GetDesktopHeight());
139 freerdp_settings_set_uint32(settings, FreeRDP_ColorDepth,
140 m_pParameter->GetColorDepth());
141
142 freerdp_settings_set_bool(settings, FreeRDP_UseMultimon,
143 m_pParameter->GetUseMultimon());
144
145 if(m_pParameter->GetReconnectInterval()) {
146 freerdp_settings_set_bool(
147 settings, FreeRDP_AutoReconnectionEnabled, true);
148 freerdp_settings_set_uint32(
149 settings,
150 FreeRDP_AutoReconnectMaxRetries,
151 m_pParameter->GetReconnectInterval());
152 }
153 else
154 freerdp_settings_set_bool(
155 settings, FreeRDP_AutoReconnectionEnabled, false);
156
157 //Load channel
158 RedirectionSound();
159 RedirectionMicrophone();
160 RedirectionDriver();
161 RedirectionPrinter();
162 RedirectionSerial();
163
164 // Set proxy
165 switch(m_pParameter->m_Proxy.GetUsedType())
166 {
167 case CParameterProxy::TYPE::None:
168 {
169 if(!m_pParameter->GetDomain().isEmpty())
170 freerdp_settings_set_string(
171 settings, FreeRDP_Domain,
172 m_pParameter->GetDomain().toStdString().c_str());
173 if(m_pParameter->m_Net.GetHost().isEmpty())
174 {
175 QString szErr;
176 szErr = tr("The server is empty, please input it");
177 qCritical(log) << szErr;
178 emit sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
179 emit sigError(-1, szErr.toStdString().c_str());
180 return OnInitReturnValue::Fail;
181 }
182 auto &net = m_pParameter->m_Net;
183 freerdp_settings_set_string(
184 settings, FreeRDP_ServerHostname,
185 net.GetHost().toStdString().c_str());
186 freerdp_settings_set_uint32(
187 settings, FreeRDP_ServerPort,
188 net.GetPort());
189
190 nRet = freerdp_client_start(pRdpContext);
191 if(nRet)
192 {
193 qCritical(log) << "freerdp_client_start fail";
194 return OnInitReturnValue::Fail;
195 }
196 break;
197 }
198#ifdef HAVE_LIBSSH
199 case CParameterProxy::TYPE::SSHTunnel:
200 {
201 // Set SSH parameters
202 QSharedPointer<CParameterChannelSSH> parameter(new CParameterChannelSSH());
203 auto &ssh = m_pParameter->m_Proxy.m_SSH;
204 parameter->setServer(ssh.GetHost());
205 parameter->setPort(ssh.GetPort());
206 auto &user = ssh.m_User;
207 parameter->SetUser(user.GetUser());
208 parameter->SetUseSystemFile(user.GetUseSystemFile());
209 if(CParameterUser::TYPE::UserPassword == user.GetUsedType()) {
210 parameter->SetAuthenticationMethod(SSH_AUTH_METHOD_PASSWORD);
211 parameter->SetPassword(user.GetPassword());
212 }
213 if(CParameterUser::TYPE::PublicKey == user.GetUsedType()) {
214 parameter->SetAuthenticationMethod(SSH_AUTH_METHOD_PUBLICKEY);
215 parameter->SetPublicKeyFile(user.GetPublicKeyFile());
216 parameter->SetPrivateKeyFile(user.GetPrivateKeyFile());
217 parameter->SetPassphrase(user.GetPassphrase());
218 }
219 auto &net = m_pParameter->m_Net;
220 parameter->SetRemoteHost(net.GetHost());
221 parameter->SetRemotePort(net.GetPort());
222
223 // Start ssh thread
224 if(!m_pThread)
225 m_pThread = new CSSHTunnelThread(parameter);
226 if(!m_pThread)
227 return OnInitReturnValue::Fail;
228 bool check = connect(m_pThread, SIGNAL(sigServer(quint16)),
229 this, SLOT(slotConnectProxyServer(quint16)));
230 Q_ASSERT(check);
231 check = connect(m_pThread, SIGNAL(sigError(int,QString)),
232 this, SIGNAL(sigError(int,QString)));
233 Q_ASSERT(check);
234 check = connect(m_pThread, SIGNAL(sigDisconnect()),
235 this, SIGNAL(sigDisconnect()));
236 Q_ASSERT(check);
237 m_pThread->start();
238 break;
239 }
240#endif
241 default:
242 break;
243 };
244
245 return OnInitReturnValue::UseOnProcess;
246}

292int CConnectFreeRDP::OnProcess()
293{
294 //qDebug(log) << "CConnectFreeRDP::OnProcess()";
295 int nRet = 0;
296 HANDLE handles[64];
297 rdpContext* pRdpContext = (rdpContext*)m_pContext;
298
299 if(nullptr == freerdp_settings_get_string(pRdpContext->settings, FreeRDP_ServerHostname))
300 {
301 return 50;
302 }
303
304 do {
305
306 DWORD nCount = 0;
307 nCount = freerdp_get_event_handles(pRdpContext, &handles[nCount],
308 ARRAYSIZE(handles) - nCount);
309 if (nCount == 0)
310 {
311 qCritical(log) << "freerdp_get_event_handles failed";
312 nRet = -2;
313 break;
314 }
315
316 handles[nCount] = m_writeEvent;
317 nCount++;
318
319 DWORD waitStatus = WaitForMultipleObjects(nCount, handles, FALSE, 500);
320
321 ResetEvent(m_writeEvent);
322
323 if (waitStatus == WAIT_FAILED)
324 {
325 qCritical(log) << "WaitForMultipleObjects: WAIT_FAILED";
326 nRet = -3;
327 break;
328 }
329
330 if(waitStatus == WAIT_TIMEOUT)
331 {
332 //qDebug(log) << "WaitForMultipleObjects timeout";
333 nRet = 0;
334 break;
335 }
336
337 if (!freerdp_check_event_handles(pRdpContext))
338 {
339 nRet = -5;
340
341 UINT32 err = freerdp_get_last_error(pRdpContext);
342 QString szErr;
343 szErr = "freerdp_check_event_handles fail.";
344 szErr += " [";
345 szErr += QString::number(err);
346 szErr += " - ";
347 szErr += freerdp_get_last_error_category(err);
348 szErr += " - ";
349 szErr += freerdp_get_last_error_name(err);
350 szErr += "] ";
351 szErr += freerdp_get_last_error_string(err);
352 qCritical(log) << szErr;
353 emit sigError(err, szErr);
354
355 // Reconnect
356 freerdp *instance = pRdpContext->instance;
357 if (client_auto_reconnect(instance))
358 {
359 nRet = 0;
360 break;
361 }
362
363 err = freerdp_get_last_error(pRdpContext);
364 szErr = "client_auto_reconnect[";
365 szErr += QString::number(err);
366 szErr += "]";
367 szErr += freerdp_get_last_error_category(err);
368 szErr += "-";
369 szErr += freerdp_get_last_error_name(err);
370 szErr += ":";
371 szErr += freerdp_get_last_error_string(err);
372 qCritical(log) << szErr;
373 emit sigError(err, szErr);
374 }
375
376#if FreeRDP_VERSION_MAJOR >= 3
377 if(freerdp_shall_disconnect_context(pRdpContext))
378#else
379 if(freerdp_shall_disconnect(pRdpContext->instance))
380#endif
381 {
382 qCritical(log) << "freerdp_shall_disconnect false";
383 nRet = -7;
384 }
385 } while(false);
386
387 return nRet;
388}

439int CConnectFreeRDP::cbClientStart(rdpContext *context)
440{
441 qDebug(log) << "CConnectFreeRdp::OnClientStart()";
442 int nRet = 0;
443
444 if (!context || !context->settings)
445 return -1;
446 freerdp* instance = freerdp_client_get_instance(context);
447 if(!instance)
448 return -2;
449 CConnectFreeRDP* pThis = ((ClientContext*)context)->pThis;
450 auto settings = context->settings;
451
452 QString szHost;
453 quint16 nPort;
454 szHost = freerdp_settings_get_string(settings, FreeRDP_ServerHostname);
455 nPort = freerdp_settings_get_uint32(settings, FreeRDP_ServerPort);
456 QString szServer;
457 auto &net = pThis->m_pParameter->m_Net;
458 szServer = net.GetHost() + ":" + QString::number(net.GetPort());
459 auto &proxy = pThis->m_pParameter->m_Proxy;
460 switch(proxy.GetUsedType()) {
461 case CParameterProxy::TYPE::SSHTunnel:
462 {
463 auto &sshNet = proxy.m_SSH;
464 szServer = szHost + ":" + QString::number(nPort)
465 + " <-> " + sshNet.GetHost() + ":" + QString::number(sshNet.GetPort())
466 + " <-> " + szServer;
467 break;
468 }
469 default:
470 break;
471 }
472
473 BOOL status = freerdp_connect(instance);
474 if (status) {
475 QString szInfo = tr("Connect to ") + szServer;
476 qInfo(log) << szInfo;
477 emit pThis->sigInformation(szInfo);
478 } else {
479 //DWORD dwErrCode = freerdp_error_info(instance);
480 UINT32 nErr = freerdp_get_last_error(context);
481
482 QString szErr;
483 szErr = tr("Connect to ") + szServer + tr(" fail.");
484 szErr += "\n[";
485 szErr += QString::number(nErr) + " - ";
486 szErr += freerdp_get_last_error_name(nErr);
487 szErr += "] ";
488 /*szErr += "[";
489 szErr += freerdp_get_last_error_category(nErr);
490 szErr += "] ";*/
491 szErr += freerdp_get_last_error_string(nErr);
492 //szErr += "]";
493
494 switch(nErr) {
495 case FREERDP_ERROR_CONNECT_LOGON_FAILURE:
496 {
497 nRet = -3;
498 QString szErr = tr("Logon to ") + szServer;
499 szErr += tr(" fail. Please check that the username and password are correct.") + "\n";
500 emit pThis->sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
501 break;
502 }
503 case FREERDP_ERROR_CONNECT_WRONG_PASSWORD:
504 {
505 nRet = -4;
506 QString szErr = tr("Logon to ") + szServer;
507 szErr += tr(" fail. Please check password are correct.") + "\n";
508 emit pThis->sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
509 break;
510 }
511 case FREERDP_ERROR_AUTHENTICATION_FAILED:
512 {
513 nRet = -5;
514 QString szErr = tr("Logon to ") + szServer;
515 szErr += tr(" authentication fail. please add a CA certificate to the store.") + "\n";
516 emit pThis->sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
517 break;
518 }
519 case FREERDP_ERROR_CONNECT_TRANSPORT_FAILED:
520 {
521 nRet = -6;
522 QString szErr = tr("Logon to ") + szServer;
523 szErr += tr(" connect transport layer fail.") + "\n\n";
524 szErr += tr("Please:") + "\n";
525 szErr += tr("1. Check for any network related issues") + "\n";
526 szErr += tr("2. Check you have proper security settings ('NLA' enabled is required for most connections nowadays)") + "\n";
527 szErr += tr("3. Check the certificate is proper (and guacd properly checks that)") + "\n";
528 emit pThis->sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
529 break;
530 }
531 case FREERDP_ERROR_SECURITY_NEGO_CONNECT_FAILED:
532 default:
533 nRet = -7;
534 emit pThis->sigShowMessageBox(tr("Error"), szErr, QMessageBox::Critical);
535 }
536
537 qCritical(log) << szErr;
538 emit pThis->sigError(nRet, szErr.toStdString().c_str());
539 }
540 return nRet;
541}

577BOOL CConnectFreeRDP::cb_pre_connect(freerdp* instance)
578{
579 qDebug(log) << "CConnectFreeRdp::cb_pre_connect()";
580 rdpChannels* channels = nullptr;
581 rdpSettings* settings = nullptr;
582 rdpContext* context = instance->context;
583
584 if (!instance || !instance->context || !instance->context->settings)
585 {
586 return FALSE;
587 }
588
589 CConnectFreeRDP* pThis = ((ClientContext*)context)->pThis;
590 if(!pThis) return FALSE;
591 settings = instance->context->settings;
592 channels = context->channels;
593
594 /* Optional OS identifier sent to server */
595#if defined (Q_OS_WIN)
596 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMajorType, OSMAJORTYPE_WINDOWS))
597 return FALSE;
598 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMinorType, OSMINORTYPE_WINDOWS_NT))
599 return FALSE;
600#elif defined(Q_OS_ANDROID)
601 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMajorType, OSMAJORTYPE_ANDROID))
602 return FALSE;
603 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMinorType, OSMINORTYPE_UNSPECIFIED))
604 return FALSE;
605#elif defined(Q_OS_IOS)
606 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMajorType, OSMAJORTYPE_IOS))
607 return FALSE;
608 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMinorType, OSMINORTYPE_UNSPECIFIED))
609 return FALSE;
610#elif defined (Q_OS_UNIX)
611 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMajorType, OSMAJORTYPE_UNIX))
612 return FALSE;
613 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMinorType, OSMINORTYPE_NATIVE_XSERVER))
614 return FALSE;
615#else
616 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMajorType, OSMAJORTYPE_UNSPECIFIED))
617 return FALSE;
618 if (!freerdp_settings_set_uint32(settings, FreeRDP_OsMinorType, OSMINORTYPE_UNSPECIFIED))
619 return FALSE;
620#endif
621
622 // Subscribe channel event
623 PubSub_SubscribeChannelConnected(instance->context->pubSub,
624 OnChannelConnectedEventHandler);
625 PubSub_SubscribeChannelDisconnected(instance->context->pubSub,
626 OnChannelDisconnectedEventHandler);
627
628#if FreeRDP_VERSION_MAJOR < 3
629 if (!freerdp_client_load_addins(channels, instance->context->settings))
630 return FALSE;
631#endif
632
633 // Check authentication parameters
634 if (freerdp_settings_get_bool(settings, FreeRDP_AuthenticationOnly))
635 {
636 /* Check +auth-only has a username and password. */
637 auto &user = pThis->m_pParameter->m_Net.m_User;
638 if(!freerdp_settings_get_string(settings, FreeRDP_Username)) {
639 if(user.GetUser().isEmpty()) {
640 if(user.GetUser().isEmpty()) {
641 // Will be call instance->Authenticate = cb_authenticate
642 qWarning(log) << "Auth-only, but no user name set. Will be call instance->Authenticate.";
643 }
644 } else
645 freerdp_settings_set_string(
646 settings, FreeRDP_Username,
647 user.GetUser().toStdString().c_str());
648 }
649 if (!freerdp_settings_get_string(settings, FreeRDP_Password)) {
650 if (user.GetPassword().isEmpty()) {
651 // Will be call instance->Authenticate = cb_authenticate
652 qWarning(log) << "auth-only, but no password set. Will be call instance->Authenticate";
653 } else
654 freerdp_settings_set_string(
655 settings, FreeRDP_Password,
656 user.GetPassword().toStdString().c_str());
657 }
658#if FreeRDP_VERSION_MAJOR >= 3
659 if (!freerdp_settings_set_bool(settings, FreeRDP_DeactivateClientDecoding, TRUE))
660 return FALSE;
661#endif
662 qInfo(log) << "Authentication only. Don't connect to X.";
663 } else if(freerdp_settings_get_bool(settings, FreeRDP_CredentialsFromStdin)){
664 // Because the pragram is GUI. so don't process it.
665 } else if(freerdp_settings_get_bool(settings, FreeRDP_SmartcardLogon)) {
666 // TODO: add FreeRDP_SmartcardLogon !
667 }
668
669 /* TODO: Check Keyboard layout
670 UINT32 rc = freerdp_keyboard_init(
671 freerdp_settings_get_uint32(settings, FreeRDP_KeyboardLayout));
672 freerdp_settings_set_uint32(settings, FreeRDP_KeyboardLayout, rc);
673 //*/
674
675 // Check desktop size, it is set in parameter
676 UINT32 width = pThis->m_pParameter->GetDesktopWidth();
677 UINT32 height = pThis->m_pParameter->GetDesktopHeight();
678 if ((width < 64) || (height < 64) ||
679 (width > 4096) || (height > 4096))
680 {
681 QString szErr = tr("Invalid dimensions:")
682 + QString::number(width)
683 + "*" + QString::number(height);
684 qCritical(log) << szErr;
685 //emit pThis->sigShowMessageBox(tr("Error"), szErr);
686 return FALSE;
687 } else {
688 qInfo(log) << "Init desktop size " << width << "*" << height;
689 }
690
691 qDebug(log)

948 case AUTH_TLS:

1027 *domain = _strdup("\0");

1030 {
1031 free(*username);
1032 qCritical(log) << "strdup failed" << status;
1033 return FALSE;
