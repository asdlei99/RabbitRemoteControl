101int CConnect::SetViewer(CFrmViewer *pView, bool bDirectConnection)
102{
103 Q_ASSERT(pView);
104 if(!pView) return -1;
105
106 if(m_pView)
107 {
108 this->disconnect(m_pView);
109 }
110
111 m_pView = pView;
112 bool check = connect(this, SIGNAL(sigSetDesktopSize(int, int)),
113 m_pView, SLOT(slotSetDesktopSize(int, int)));
114 Q_ASSERT(check);
115 check = connect(this, SIGNAL(sigServerName(const QString&)),
116 m_pView, SLOT(slotSetName(const QString&)));
117 Q_ASSERT(check);
118
119 check = connect(this, SIGNAL(sigUpdateRect(const QRect&, const QImage&)),
120 m_pView, SLOT(slotUpdateRect(const QRect&, const QImage&)));
121 Q_ASSERT(check);
122 check = connect(this, SIGNAL(sigUpdateRect(const QImage&)),
123 m_pView, SLOT(slotUpdateRect(const QImage&)));
124 Q_ASSERT(check);
125 check = connect(this, SIGNAL(sigUpdateCursor(const QCursor&)),
126 m_pView, SLOT(slotUpdateCursor(const QCursor&)));
127 Q_ASSERT(check);
128 check = connect(this, SIGNAL(sigUpdateCursorPosition(const QPoint&)),
129 m_pView, SLOT(slotUpdateCursorPosition(const QPoint&)));
130 Q_ASSERT(check);
131 check = connect(this, SIGNAL(sigUpdateLedState(unsigned int)),
132 m_pView, SLOT(slotUpdateLedState(unsigned int)));
133 Q_ASSERT(check);
134
135 if(bDirectConnection)
136 {
137 /* \~chinese 因为连接可能是在另一个线程中的非Qt事件处理，
138 * 它可能会阻塞线程，那会导致键盘、鼠标事件延迟，
139 * 所以这里用 Qt::DirectConnection
140 * \~english Because the connection may be a non-Qt event processing in another thread,
141 * it may block the thread, which will cause the keyboard and mouse events to be delayed.
142 * So here use Qt::DirectConnection
143 */
144 check = connect(m_pView, SIGNAL(sigMousePressEvent(QMouseEvent*, QPoint)),
145 this, SLOT(slotMousePressEvent(QMouseEvent*, QPoint)),
146 Qt::DirectConnection);
147 Q_ASSERT(check);
148 check = connect(m_pView, SIGNAL(sigMouseReleaseEvent(QMouseEvent*, QPoint)),
149 this, SLOT(slotMouseReleaseEvent(QMouseEvent*, QPoint)),
150 Qt::DirectConnection);
151 Q_ASSERT(check);
152 check = connect(m_pView, SIGNAL(sigMouseMoveEvent(QMouseEvent*, QPoint)),
153 this, SLOT(slotMouseMoveEvent(QMouseEvent*, QPoint)),
154 Qt::DirectConnection);
155 Q_ASSERT(check);
156 check = connect(m_pView, SIGNAL(sigWheelEvent(QWheelEvent*, QPoint)),
157 this, SLOT(slotWheelEvent(QWheelEvent*, QPoint)),
158 Qt::DirectConnection);
159 Q_ASSERT(check);
160 check = connect(m_pView, SIGNAL(sigKeyPressEvent(QKeyEvent*)),
161 this, SLOT(slotKeyPressEvent(QKeyEvent*)),
162 Qt::DirectConnection);
163 Q_ASSERT(check);
164 check = connect(m_pView, SIGNAL(sigKeyReleaseEvent(QKeyEvent*)),
165 this, SLOT(slotKeyReleaseEvent(QKeyEvent*)),
166 Qt::DirectConnection);
167 Q_ASSERT(check);
168 } else {
169 check = connect(m_pView, SIGNAL(sigMousePressEvent(QMouseEvent*, QPoint)),
170 this, SLOT(slotMousePressEvent(QMouseEvent*, QPoint)));
171 Q_ASSERT(check);
172 check = connect(m_pView, SIGNAL(sigMouseReleaseEvent(QMouseEvent*, QPoint)),
173 this, SLOT(slotMouseReleaseEvent(QMouseEvent*, QPoint)));
174 Q_ASSERT(check);
175 check = connect(m_pView, SIGNAL(sigMouseMoveEvent(QMouseEvent*, QPoint)),
176 this, SLOT(slotMouseMoveEvent(QMouseEvent*, QPoint)));
177 Q_ASSERT(check);
178 check = connect(m_pView, SIGNAL(sigWheelEvent(QWheelEvent*, QPoint)),
179 this, SLOT(slotWheelEvent(QWheelEvent*, QPoint)));
180 Q_ASSERT(check);
181 check = connect(m_pView, SIGNAL(sigKeyPressEvent(QKeyEvent*)),
182 this, SLOT(slotKeyPressEvent(QKeyEvent*)));
183 Q_ASSERT(check);
184 check = connect(m_pView, SIGNAL(sigKeyReleaseEvent(QKeyEvent*)),
185 this, SLOT(slotKeyReleaseEvent(QKeyEvent*)));
186 Q_ASSERT(check);
187 }
188
189 return 0;
190}

212void CConnect::slotTimeOut()
213{
214 //qDebug(log) << "CConnect::slotTimeOut()";
215 try {
216 // >= 0: continue. Call interval
217 // = -1: stop
218 // < -1: error
219 int nTime = OnProcess();
220 if(nTime >= 0)
221 {
222 QTimer::singleShot(nTime, this, SLOT(slotTimeOut()));
223 return;
224 }
225 qCritical(log) << "Process fail or stop:" << nTime;
226 if(nTime < -1)
227 emit sigError(nTime, "Process fail or stop");
228 } catch(std::exception e) {
229 qCritical(log) << "Process fail:" << e.what();
230 emit sigError(-2, e.what());
231 } catch (...) {
232 qCritical(log) << "Process fail";
233 emit sigError(-3, "Process fail");
234 }
235
236 // Error or stop, must notify user disconnect it
237 emit sigDisconnect();
238}

336bool CConnect::event(QEvent *event)
337{
338 //qDebug(log) << "CConnect::event" << event;
339 switch (event->type()) {
340 case QEvent::MouseButtonPress:
341 case QEvent::MouseButtonDblClick:
342 mousePressEvent((QMouseEvent*)event);
343 break;
344 case QEvent::MouseButtonRelease:
345 mouseReleaseEvent((QMouseEvent*)event);
346 break;
347 case QEvent::MouseMove:
348 mouseMoveEvent((QMouseEvent*)event);
349 break;
350 case QEvent::Wheel:
351 wheelEvent((QWheelEvent*)event);
352 break;
353 case QEvent::KeyPress:
354 keyPressEvent((QKeyEvent*)event);
355 break;
356 case QEvent::KeyRelease:
357 keyReleaseEvent((QKeyEvent*)event);
358 break;
359 default:
360 return QObject::event(event);
361 }
362
363 event->accept();
364 return true;
365}
